CREATE TABLE IF NOT EXISTS promotionalproduct (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(255) NOT NULL,
    descripcion TEXT,
    precio_en_promocion DECIMAL(10, 2) NOT NULL,
    fecha_inicio_promocion DATE NOT NULL,
    fecha_fin_promocion DATE NOT NULL,
    activo BOOLEAN DEFAULT TRUE,
    id_producto INT,
    FOREIGN KEY (id_producto) REFERENCES products(id),
    campo_adicional_2 INT
);
